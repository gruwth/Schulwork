    document.addEventListener('DOMContentLoaded', function() {
        const colors = ['#000', '#6e44ff', '#ff77e9']; // Colors: black, purple, pink
        let currentColorIndex = 0;

        function changeColor() {
            // Move to the next color in the array
            currentColorIndex = (currentColorIndex + 1) % colors.length;
            
            // Update the background color with a smooth transition
            document.body.style.backgroundColor = colors[currentColorIndex];
        }

        // Change color every 6 seconds to account for transition time
        setInterval(changeColor, 1000); // Adjust timing as needed
    });
</script>
